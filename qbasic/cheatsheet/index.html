QBasic Cheatsheet


' this line is a comment. The computer ignores it. 

name$ = "Cake"
PRINT LCASE$(name$) 'LCASE$ makes it lowercase
PRINT UCASE$(name$) 'UCASE$ makes it capitals

coins = 3
IF coins > 1 THEN
  PRINT "many coins"
ELSE
  PRINT "that's unexpected."
END IF

DO
  PRINT "this will loop forever"
LOOP

' Press Ctrl+Break to stop a program that's 
'stuck in a loop. This doesn't work in our online version :(

X = 0
DO
  X = X + 1
LOOP UNTIL X = 4

FOR I = 1 TO 100
  PRINT I
NEXT I

FOR I = 1 TO 100 STEP 5
  PRINT I
NEXT I

' Print on the same line by using a ;
PRINT "line 1";
PRINT "still line 1"
PRINT "line 2."

' Subs let you make your own commands.
' First you write what the command should do:
SUB PrintCoins
  COLOR 15
  PRINT "You have"; coins; "coins left."
  COLOR 4
  IF coins = 10 THEN PRINT "That's enough for a hat!"
END SUB

' This will take you to a new tab, kind of.
' Press F2 to switch between tabs and find your old code

' Now you can run all that code just by writing this:
PrintCoins

' You can pass arguments to a sub
SUB DrawCircle (radius)
  '....
END SUB

DrawCircle 10
